// @lc code=start

enum CarType {
    "big" = 1,
    "medium" = 2,
    "small" = 3,
}
class ParkingSystem {
    big: number = 0;
    medium: number = 0;
    small: number = 0;

    constructor(big: number, medium: number, small: number) {
        this.big = big;
        this.medium = medium;
        this.small = small;
    }

    addCar(carType: number): boolean {
        return --this[CarType[carType]] >= 0;
    }
}
